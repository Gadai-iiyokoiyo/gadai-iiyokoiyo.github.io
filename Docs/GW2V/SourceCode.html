import random
import json
import sys
max_rs=100
if(len(sys.argv)==1):
    WordVec={"PlayStation":random.uniform(-max_rs,max_rs),
    "Sony":random.uniform(0,max_rs),"Nintendo":random.uniform(0,max_rs),"花札":random.uniform(0,max_rs),"100円":random.uniform(0,max_rs)
    ,"王":random.uniform(0,max_rs),"男":random.uniform(0,max_rs),"女":random.uniform(0,max_rs),
    "子供":random.uniform(0,max_rs),"Garry'smod":random.uniform(0,max_rs),"Sandbox":random.uniform(0,max_rs),"Battleroyal":random.uniform(0,max_rs),
    "FPS":random.uniform(0,max_rs),"Valve":random.uniform(0,max_rs),"RPG":random.uniform(0,max_rs)}
    for i in range(120):
        WordVec["GameConsole"]=WordVec["PlayStation"]-WordVec["Sony"]
        WordVec["PlayStation"]=WordVec["GameConsole"]+WordVec["Sony"]
        WordVec["NintendoSwitch"]=WordVec["GameConsole"]+WordVec["Nintendo"]
        WordVec["GameConsole"]=WordVec["NintendoSwitch"]-WordVec["Nintendo"]
        WordVec["花札屋"]=WordVec["Nintendo"]-WordVec["GameConsole"]
        WordVec["店"]=WordVec["花札屋"]-WordVec["花札"]
        WordVec["花札"]=WordVec["花札屋"]-WordVec["店"]
        WordVec["女王"]=WordVec["王"]+WordVec["女"]
        WordVec["王子"]=WordVec["王"]+WordVec["子供"]
        WordVec["王女"]=WordVec["王子"]+WordVec["女"]
        WordVec["100円ショップ"]=WordVec["店"]+WordVec["100円"]
        WordVec["Halflife2"]=WordVec["Garry'smod"]-WordVec["Sandbox"]
        WordVec["Garry'smod"]=WordVec["Halflife2"]+WordVec["Sandbox"]
        WordVec["CounterStrike"]=WordVec["Halflife2"]+WordVec["Battleroyal"]
        WordVec["Halflife2"]=WordVec["CounterStrike"]-WordVec["Battleroyal"]
        WordVec["Garry'smod"]=WordVec["CounterStrike"]+WordVec["Sandbox"]
        WordVec["Halflife2"]=WordVec["CounterStrike"]+WordVec["RPG"]
        WordVec["Valve"]=WordVec["Halflife2"]+WordVec["FPS"]

    json.dump(WordVec, open("WordVec.json","w",encoding="utf-8"), indent=4)
else:
    WordVec=json.load(open(sys.argv[1],"r",encoding="utf-8"))
reverse_WordVec={v:k for k,v in WordVec.items()}
while True:
    try:
        originalText=input(">").replace(" ","")
        if("+" in originalText):
            originalText_Splits=originalText.split("+")
            a=originalText_Splits[0]
            b=originalText_Splits[1]
            try:
                print(reverse_WordVec[WordVec[a]+WordVec[b]])
            except:
                print(f"エラーが発生しました。:{WordVec[a]+WordVec[b]}")
        elif("-" in originalText):
            originalText_Splits=originalText.split("-")
            a=originalText_Splits[0]
            b=originalText_Splits[1]
            try:
                print(reverse_WordVec[WordVec[a]-WordVec[b]])
            except:
                print(f"エラーが発生しました。:{WordVec[a]-WordVec[b]}")
        else:
            print("+ もしくは - を書いてください。")
    except IndexError:
        print("エラーが発生しました。マイナス記号もしくはプラス記号の前と後ろに単語を書いてください。")
    except KeyError:
        print("エラーが発生しました。辞書を参照してみましたが、そのような単語が見つからないためです。")
